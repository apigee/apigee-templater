Getting Started

1. Authentication

Sign up and get your API key from your dashboard. Keep it secure and use it as an environment variable (`GEMINI_API_KEY`).

2. Install SDK

Choose your language below and install the respective SDK to begin calling the endpoint.

Code Snippets 

    # Install the SDK
    pip install gemini-ai

    import gemini_ai as ai

    client = ai.Client(api_key="YOUR_API_KEY")
    response = client.generate(
        model="flash",
        prompt="Write a 3-line poem about the sea."
    )
    print(response.text)

    // Install the SDK
    go get github.com/gemini/ai

    package main

    import (
        "fmt"
        "gemini/ai"
    )

    func main() {
        client := ai.NewClient("YOUR_API_KEY")
        resp, _ := client.Generate(
            "flash",
            "Write a 3-line poem about the sea.",
        )
        fmt.Println(resp.Text)
    }

    // Install the SDK
    npm install @gemini/sdk

    import { GeminiClient } from "@gemini/sdk";

    const client = new GeminiClient({ apiKey: "YOUR_API_KEY" });

    async function run() {
      const response = await client.generate({
        model: "flash",
        prompt: "Write a 3-line poem about the sea.",
      });
      console.log(response.text);
    }
    run();
